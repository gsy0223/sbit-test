    <script>
        let currentQuestion = 0;
        const questions = document.querySelectorAll('.question');
        const progressBar = document.getElementById('progressBar');
        const questionContainer = document.getElementById('questionContainer');
        const resultContainer = document.getElementById('resultContainer');
        const personalityTypeEl = document.getElementById('personalityType');
        const personalityDescEl = document.getElementById('personalityDesc');

        const results = {

            'INTP': '好奇独立的逻辑学家，热爱思考和分析，对世界充满疑问，是天生的思想家。',

            'ENFP': '热情好奇的竞选者，充满创意和灵感，总能给身边的人带来活力。',
            'ENTP': '聪明机智的辩论家，喜欢挑战和辩论，思维活跃，总能提出新奇的想法。',
            'ESTJ': '务实高效的管理者，擅长组织和管理，是团队里靠谱的领导者。',
            'ESFJ': '热情友善的执政官，乐于助人，重视和谐，总能照顾好身边的每一个人。',
            'ENFJ': '富有魅力的教育家，善于激励他人，充满同理心，是天生的领导者。',
            'ENTJ': '果断自信的指挥官，目标导向，擅长规划和领导，总能带领团队达成目标。'
        };

        document.querySelectorAll('.option').forEach(option => {
            option.addEventListener('click', function() {
                document.querySelectorAll('.option').forEach(opt => opt.classList.remove('selected'));
                this.classList.add('selected');

                const type = this.dataset.value;
                if (type === 'I' || type === 'E') localStorage.setItem('IE', type);
                if (type === 'S' || type === 'N') localStorage.setItem('SN', type);
                if (type === 'T' || type === 'F') localStorage.setItem('TF', type);
                if (type === 'J' || type === 'P') localStorage.setItem('JP', type);

                setTimeout(() => {
                    nextQuestion();
                }, 500);
            });
        });

        function nextQuestion() {
            questions[currentQuestion].classList.remove('active');
            currentQuestion++;

            progressBar.style.width = (currentQuestion / questions.length) * 100 + '%';

            if (currentQuestion < questions.length) {
                questions[currentQuestion].classList.add('active');
            } else {
                showResult();
            }
        }

        function showResult() {
            questionContainer.style.display = 'none';
            resultContainer.style.display = 'block';

            const ie = localStorage.getItem('IE') || 'I';
            const sn = localStorage.getItem('SN') || 'S';
            const tf = localStorage.getItem('TF') || 'T';
            const jp = localStorage.getItem('JP') || 'J';
            const type = ie + sn + tf + jp;

            personalityTypeEl.textContent = type;
            personalityDescEl.textContent = results[type];
        }

        function restartTest() {
            currentQuestion = 0;
            progressBar.style.width = '0%';
            questionContainer.style.display = 'block';
            resultContainer.style.display = 'none';
            questions.forEach((q, i) => {
                if (i === 0) q.classList.add('active');
                else q.classList.remove('active');
            });
            document.querySelectorAll('.option').forEach(opt => opt.classList.remove('selected'));
            localStorage.clear();
        }
    </script>
